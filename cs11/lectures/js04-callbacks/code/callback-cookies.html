Asynchronous Cookie Baking Demo

Seconds passed: 0

function bakeCookies() {
  let cookieState = "raw cookie dough";
  setTimeout(function() {
    console.log("Step 1: Adding chocolate chips (1s task)");
  }, 1000);
  console.log("Step 2: Starting oven...");

  setTimeout(function() {
    console.log("Step 3: Oven ready (4s task)");
  }, 4000);
  console.log("Step 4: Putting cookie dough in oven");

  setTimeout(function() {
    console.log("Step 5: Baking finished! (2s task)");
    cookieState = "golden-brown cookie";
  }, 2000);
  console.log("Step 6: Your " + cookieState + " is ready!");
}
